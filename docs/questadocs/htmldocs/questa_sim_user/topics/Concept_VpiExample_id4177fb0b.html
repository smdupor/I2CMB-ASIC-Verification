VPI Example

The following example is a trivial, but complete VPI application. A general VPI example can be found in <install_dir>/questasim/examples/verilog/vpi.

hello.c:

	#include "vpi_user.h"
	static PLI_INT32 hello(PLI_BYTE8 * param)
	{
		vpi_printf( "Hello world!\n" );
		return 0;
	}
	void RegisterMyTfs( void )
	{
		s_vpi_systf_data systf_data;
		vpiHandle systf_handle;
		systf_data.type        = vpiSysTask;
		systf_data.sysfunctype = vpiSysTask;
		systf_data.tfname      = "$hello";
		systf_data.calltf      = hello;
		systf_data.compiletf   = 0;
		systf_data.sizetf      = 0;
		systf_data.user_data   = 0;
		systf_handle = vpi_register_systf( &systf_data );
		vpi_free_object( systf_handle );
	}
	void (*vlog_startup_routines[])() = {
		RegisterMyTfs,
		0
	};
hello.v:
	module hello;
		initial $hello;
	endmodule
Compile the Verilog code:
	% vlib work
	% vlog hello.v
Simulate the design:
	% vsim -c -pli hello.sl hello
	# Loading work.hello
	# Loading ./hello.sl
	VSIM 1> run -all
	# Hello world!
	VSIM 2> quit
