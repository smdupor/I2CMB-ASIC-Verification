mti_GetWlfFilename()

Gets the name of the waveform logfile (.wlf).

Syntax

filename = mti_GetWlfFilename()

Arguments

None

Return Values

Name

Type

Description

filename

char *

A pointer to the name of the waveform logfile (.wlf)

Description

mti_GetWlfFilename() returns the name of the waveform logfile (.wlf). You must not free the returned pointer.

Examples

FLI code

#include <mti.h>

void initForeign(
  mtiRegionIdT       region,   /* The ID of the region in which this     */
                               /* foreign architecture is instantiated.  */
  char              *param,    /* The last part of the string in the     */
                               /* foreign attribute.                     */
  mtiInterfaceListT *generics, /* A list of generics for the foreign model.*/
  mtiInterfaceListT *ports     /* A list of ports for the foreign model.   */
)
{
  char * filename;

  filename = mti_GetWlfFilename();
  mti_PrintFormatted( "WLF filename = %s\n", filename );
}
